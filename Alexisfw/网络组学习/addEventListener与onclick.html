addEventListener vs onclick

理解JavaScript中两种事件处理方式的区别

addEventListener

  • 可以添加多个事件处理程序
  • 可以控制事件传播阶段（捕获/冒泡）
  • 支持更多事件类型
  • 需要使用removeEventListener移除
  • IE9+支持
  • 更灵活，推荐用于复杂应用

onclick

  • 只能有一个事件处理程序
  • 只能在冒泡阶段处理事件
  • 支持基本事件类型
  • 通过设置为null移除
  • 所有浏览器支持
  • 简单直接，适合简单场景

代码示例

// addEventListener 示例
document.getElementById('addEventBtn').addEventListener('click', function() {
    console.log('第一个处理程序');
});

document.getElementById('addEventBtn').addEventListener('click', function() {
    console.log('第二个处理程序');
});

// onclick 示例
document.getElementById('onclickBtn').onclick = function() {
    console.log('第一个onclick处理程序');
};

// 这会覆盖之前的处理程序
document.getElementById('onclickBtn').onclick = function() {
    console.log('第二个onclick处理程序');
};

// 事件传播控制
document.getElementById('outerDiv').addEventListener('click', function() {
    console.log('外部元素 - 冒泡阶段');
}, false); // false = 冒泡阶段

document.getElementById('innerDiv').addEventListener('click', function() {
    console.log('内部元素 - 捕获阶段');
}, true); // true = 捕获阶段

// 移除事件处理程序
function myEventHandler() {
    console.log('事件被处理');
}

document.getElementById('addHandlerBtn').addEventListener('click', function() {
    document.getElementById('testHandlerBtn').addEventListener('click', myEventHandler);
    console.log('事件处理程序已添加');
});

document.getElementById('removeHandlerBtn').addEventListener('click', function() {
    document.getElementById('testHandlerBtn').removeEventListener('click', myEventHandler);
    console.log('事件处理程序已移除');
});
